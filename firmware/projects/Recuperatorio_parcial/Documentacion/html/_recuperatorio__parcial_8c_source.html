56static void MedirDistanciaTask(void *pvParameter)
57{
58 while (true)
59 {
60 /* Medir la distancia en cm */
61 distancia = HcSr04ReadDistanceInCentimeters();
62
63 /* Si está cerca */
64 if (distancia < DISTANCIA_MINIMA)
65 {
66 LedOn(LED_1); /* Prende LED1, apaga LED2 y LED3 */
67 LedOff(LED_2);
68 LedOff(LED_3);
69 }
70 /* Si está lejos */
71 else if (distancia > DISTANCIA_MAXIMA)
72 {
73 LedOff(LED_1);
74 LedOff(LED_2);
75 LedOn(LED_3); /* Prende LED3, apaga LED1 y LED2 */
76 }
77 /* Si está dentro del rango */
78 else
79 {
80 LedOff(LED_1);
81 LedOn(LED_2); /* Prende LED2, apaga LED1 y LED3 */
82 LedOff(LED_3);
83 }
84 LedsOffAll(); // Apago todos los LEDs después de la medición de distancia
85
86 vTaskDelay(pdMS_TO_TICKS(TIEMPO_MEDICION_PRESENCIA)); // Cada 'TIEMPO_MEDICION_PRESENCIA' milisegundos
87 }
88}

105static void MedirTemperaturaTask(void *pvParameter)
106{
107 while (true)
108 {
109 /* Si está dentro del rango */
110 if ((distancia > DISTANCIA_MINIMA) & (distancia < DISTANCIA_MAXIMA))
111 {
112 /* Lee el valor de la entrada analógica CH1 (en mV) y lo almacena en Vin */
113 AnalogInputReadSingle(CH1, &Vin);
114 for (int i = 0; i < 10; i++)
115 {
116 /*Sumar las 10 temperaturas*/
117 temperatura += Vin * TEMPERATURA_MAXIMA / VREF;
118 // 3.3V representan 50°C
119 // Vin representa a temp °C --> temp = Vin * 50°C / 3.3V
120 }
121 promedio_temperatura = temperatura / 10; // Calculo promedio
122 EscribirEnMonitor(); // Llamo a la función para que escriba en monitor
123 }
124
125 /* Si la persona se mueve más de 140 cm */
126 if (distancia > DISTANCIA_NUEVA_MEDICION)
127 {
128 temperatura = 0; // Vuelvo a inicializar en 0 para otro ciclo de lectura
129 promedio_temperatura = 0;
130 }
131
132 vTaskDelay(pdMS_TO_TICKS(TIEMPO_MEDICION_TEMPERATURA)); // Cada 'TIEMPO_MEDICION_TEMPERATURA' milisegundos
133 }
134}
