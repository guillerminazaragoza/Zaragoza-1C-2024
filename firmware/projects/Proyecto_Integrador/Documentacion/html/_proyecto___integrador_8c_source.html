76void leerDatos(uint8_t *data, uint8_t length)
77{
78 uint8_t i = 1;
79
80 if (data[0] == 'O')
81 {
82 encendido = true;
83 printf("Se prende\n");
84 }
85
86 else if (data[0] == 'o')
87 {
88 encendido = false;
89 printf("Se apaga\n");
90 }
91
92 if (encendido)
93 {
94 if (data[0] == 'M')
95 {
96 printf("Lee tiempo: ");
97 /* El cuadro de texto envía los datos con el formato "M" + value + "m" */
98 while (data[i] != 'm')
99 {
100 /* Convertir el valor ASCII a un valor entero */
101 tiempoIdeal = tiempoIdeal * 10;
102 tiempoIdeal = tiempoIdeal + (data[i] - '0');
103 i++;
104 }
105 printf("%.2f segundos\n", tiempoIdeal);
106
107 tiempoIdealMS = tiempoIdeal * 1000; // paso a milisegundos el dato leido
108 tiempoLED = tiempoIdealMS / NUM_LEDS;
109
110 char cadena_tiempo[128];
111 strcpy(cadena_tiempo, "");
112 sprintf(cadena_tiempo, "*d%.2f\r\n*", tiempoIdeal);
113 BleSendString(cadena_tiempo);
114
115 tiempoIdeal = 0;
116 }
117 }

161static void ControlHyT_Task()
162{
163 uint8_t factor = 1; // Factor de corrección del tiempo de encendido entre cada LED
164
165 while (true)
166 {
167 vTaskDelay(pdMS_TO_TICKS(300)); // Sin esto no anda
168
169 float tiempoTotal = 0;
170 float tiempoProm = 0;
171 float velocidad = 0;
172
173 if (encendido)
174 {
175 for (int i = 0; i < NUM_LEDS; i++)
176 {
177 if (((temperatura > UT_BAJO) && (temperatura < UT_ALTO)) && ((humedad > UH_BAJO) && (humedad < UH_ALTO)))
178 {
179 factor = 1;
180 }
181 else if ((temperatura < UT_BAJO) || (temperatura > UT_ALTO))
182 {
183 factor = 1.5;
184 }
185 else if ((humedad < UH_BAJO) || (humedad > UH_ALTO))
186 {
187 factor = 2; // Las condiciones de humedad afectan más que las de temperatura
188 }
189 if (i % 2 == 0)
190 {
191 NeoPixelSetPixel(i, NEOPIXEL_COLOR_RED);
192 vTaskDelay(pdMS_TO_TICKS(tiempoLED * factor));
193 tiempoTotal += (tiempoLED * factor); // Calculo el tiempo total real (va a cambiar si fue afectado por las condiciones ambientales)
194 NeoPixelSetPixel(i, NeoPixelRgb2Color(0, 0, 0));
195 }
196 else
197 {
198 NeoPixelSetPixel(i, NEOPIXEL_COLOR_BLUE);
199 vTaskDelay(pdMS_TO_TICKS(tiempoLED * factor));
200 tiempoTotal += (tiempoLED * factor);
201 NeoPixelSetPixel(i, NeoPixelRgb2Color(0, 0, 0));
202 }
203 }
204 NeoPixelAllOff();
205
206 tiempoTotal *= 0.001; // al valor en milisegundos lo muestro como segundos
207 tiempoProm = tiempoTotal / NUM_LEDS;
208 velocidad = distancia / tiempoTotal;
209
210 /* Mostrar los datos por pantalla */
211 printf("Tiempo promedio (por LED): %.2f s\n", tiempoProm);
212 printf("Tiempo total: %.2f s\n", tiempoTotal);
213 printf("Velocidad en la vuelta: %.5f m/s\n\n", velocidad);
214
215 /* Mostrar los datos por la aplicación */
216 char cadena_vel[128];
217 strcpy(cadena_vel, "");
218 sprintf(cadena_vel, "*V%.2f\n*", velocidad);
219 BleSendString(cadena_vel);
220
221 char cadena_tiempoProm[128];
222 strcpy(cadena_tiempoProm, "");
223 sprintf(cadena_tiempoProm, "*P%.2f\n*", tiempoProm);
224 BleSendString(cadena_tiempoProm);
225
226 char cadena_tiempoTotal[128];
227 strcpy(cadena_tiempoTotal, "");
228 sprintf(cadena_tiempoTotal, "*R%.2f\n*", tiempoTotal);
229 BleSendString(cadena_tiempoTotal);
230 }
